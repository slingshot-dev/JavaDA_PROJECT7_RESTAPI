package com.nnk.springboot.domain;

import javax.persistence.*;
import javax.validation.constraints.NotBlank;

import java.sql.Timestamp;

@Entity
@Table(name = "rule")

    @NotBlank

    @NotBlank

    @Column(name = "template")

    @Column(name = "sql_str")

        this.sqlStr = sqlStr;
